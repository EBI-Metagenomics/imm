function(imm_add_test name)
    add_executable(test_${name} src/${name}.c)
    target_link_libraries(test_${name} PRIVATE IMM::imm)
    target_link_libraries(test_${name} PRIVATE IMM::warning)
    add_test(NAME ${name} COMMAND test_${name})
endfunction()
